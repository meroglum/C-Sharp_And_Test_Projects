WebDriverExtension

	public static void EnterText(this IWebDriver driver, By locator, string value)
	{
		IWebElement ele = driver.FindElement(locator);
		if (ele != null && ele.Enabled)
		{
			ele.Clear();
			ele.SendKeys(value);
		}
	}
